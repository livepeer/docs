	}'

```go Go
package main

import (
	"context"
	"log"
	"livepeer"
)

func main() {
	client := livepeer.New(
		livepeer.WithSecurity("<api-key>"),
	)

	ctx := context.Background()
	id := "08eeeda4-681f-46e5-a760-63d6cac91d91"

	requestBody := components.MultistreamTargetPatchPayload{
		Name: "an updated name",
		Disabled: true,
	}

res, err := client.MultistreamTarget.Update(ctx, id, requestBody)

	if err != nil {
		log.Fatal(err)
	}

	if res.Stream != nil {
		// Handle the response (res.Stream)
	}
}
```
