```javascript JavaScript
import { Livepeer } from "livepeer";

const apiToken = "YOUR_API_TOKEN";
const assetId = "ASSET_ID";
const timeStep = "day";
const breakdownBy = ["timezone"];

const livepeer = new Livepeer({
  apiKey: apiToken,
});

const queryParams = {
  assetId: assetId,
  timeStep: timeStep,
  breakdownBy: breakdownBy,
};

livepeer.metrics
  .getCreatorViewership(queryParams)
  .then((response) => {
    console.log("Views by creator:", response);
  })
  .catch((error) => {
    console.error("Error fetching views by creator:", error);
  });
```

```go Go
// Go
package main

import(
  "context"
  "log"
  "livepeer"
  "livepeer/models/components"
  "livepeer/models/operations"
  "livepeer/types"
)

func main() {
    lpClient := livepeer.New(
        livepeer.WithSecurity("<api-key>"),
    )

    ctx := context.Background()
    res, err := lpClient.Metrics.GetCreatorViewership(ctx, operations.GetCreatorMetricsRequest{
        BreakdownBy: []operations.QueryParamBreakdownBy{
            operations.QueryParamBreakdownByDeviceType,
        },
    })
    if err != nil {
        log.Fatal(err)
    }

    if res.Data != nil {
        // handle response
    }
}
```
