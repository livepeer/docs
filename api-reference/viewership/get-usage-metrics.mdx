<RequestExample>

```bash cURL
curl --request GET \
  --url https://livepeer.studio/api/data/usage/query \
  --header 'Authorization: <api-key>'
```

```javascript JavaScript
import { Livepeer } from "livepeer";

const apiToken = "YOUR_API_TOKEN";
const breakdownBy = ["device", "browser"];

const livepeer = new Livepeer({
  apiKey: apiToken,
});

const queryParams = {
  breakdownBy: breakdownBy,
};

livepeer.metrics
  .getPublicTotalViews(queryParams)
  .then((response) => {
    console.log("Views:", response);
  })
  .catch((error) => {
    console.error("Error fetching views:", error);
  });
```

```go Go
// Go
package main

import(
  "context"
  "log"
  "livepeer"
  "livepeer/models/components"
)

func main() {
    lpClient := livepeer.New(
        livepeer.WithSecurity("<api-key>"),
    )

    var playbackID string = "PLAYBACK_ID"

    ctx := context.Background()

    res, err := lpClient.Metrics.GetPublicTotalViews(ctx, playbackID)
    if err != nil {
        log.Fatal(err)
    }

    if res.Data != nil {
        // handle response
    }
}
```
