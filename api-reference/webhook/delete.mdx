```bash cURL
curl --request DELETE \
  --url https://livepeer.studio/api/webhook/{id} \
  --header 'Authorization: Bearer <api-key>'
```

```javascript JavaScript
import { Livepeer } from "livepeer";

const livepeer = new Livepeer({
  apiKey: "<api-key>",
});

const id = "08eeeda4-681f-46e5-a760-63d6cac91d91";

await livepeer.webhook.delete(id);
```

```go Go
package main

import (
	"context"
	"log"
	"livepeer"
)

func main() {
	client := livepeer.New(
		livepeer.WithSecurity("<api-key>"),
	)

	ctx := context.Background()
	id := "08eeeda4-681f-46e5-a760-63d6cac91d91"

	res, err := client.Webhook.Delete(ctx, id)

	if err != nil {
		log.Fatal(err)
	}

	if res.Stream != nil {
		// Handle the response (res.Stream)
	}
}
```
